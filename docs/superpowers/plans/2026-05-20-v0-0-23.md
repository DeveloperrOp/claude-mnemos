```python
def recover_zombie_running(self) -> int:
    with self._tx_lock:
        try:
            self._conn.execute("BEGIN IMMEDIATE")
            cur = self._conn.execute(
                "SELECT id FROM jobs WHERE status='running'"
            )
            ids = [row["id"] for row in cur.fetchall()]
            if not ids:
                self._conn.execute("COMMIT")
                return 0
            placeholders = ",".join("?" for _ in ids)
            self._conn.execute(
                f"UPDATE jobs SET status='pending', attempts=0 WHERE id IN ({placeholders})",
                ids,
            )
            self._conn.execute("COMMIT")
            return len(ids)
        except Exception:
            with contextlib.suppress(Exception):
                self._conn.execute("ROLLBACK")
            raise
```

```python
import hashlib
import tarfile as _tarfile
from pydantic import BaseModel


class RestorePreview(BaseModel):
    snapshot_name: str
    snapshot_timestamp: str
    snapshot_kind: str
    snapshot_file_count: int
    vault_file_count: int
    will_create: list[str]
    will_delete: list[str]
    will_overwrite: list[str]
    unchanged_count: int
    sample_limit: int
    truncated: bool


_SNAPSHOT_SKIP_PREFIXES = (".snapshots/", ".manifest.json", ".inject-metrics.json")


def _iter_vault_files(vault: Path) -> list[tuple[str, Path]]:
    """Walk vault returning (vault-relative-posix, abs_path) skipping hidden dirs."""
    results: list[tuple[str, Path]] = []
    for p in vault.rglob("*"):
        if not p.is_file():
            continue
        rel = p.relative_to(vault).as_posix()
        if any(rel.startswith(skip) for skip in _SNAPSHOT_SKIP_PREFIXES):
            continue
        if any(part.startswith(".") for part in p.relative_to(vault).parts):
            continue
        results.append((rel, p))
    return results


def _file_sha256(path: Path) -> str | None:
    try:
        h = hashlib.sha256()
        with open(path, "rb") as f:
            for chunk in iter(lambda: f.read(65536), b""):
                h.update(chunk)
        return h.hexdigest()
    except OSError:
        return None


def compute_restore_preview(
    vault: Path,
    snapshot_name: str,
    *,
    sample_limit: int = 20,
) -> RestorePreview:
    snap_dir = vault / ".snapshots"
    snap_path = snap_dir / f"{snapshot_name}.tar.gz"
    if not snap_path.exists():
        raise FileNotFoundError(f"Snapshot not found: {snapshot_name}")

    # Read snapshot manifest
    snapshot_files: dict[str, bytes] = {}
    with _tarfile.open(snap_path, "r:gz") as tar:
        for member in tar.getmembers():
            if member.isfile():
                f = tar.extractfile(member)
                if f is not None:
                    snapshot_files[member.name] = f.read()

    # Read vault files
    vault_files: dict[str, Path] = dict(_iter_vault_files(vault))

    snap_keys = set(snapshot_files.keys())
    vault_keys = set(vault_files.keys())

    will_create_all: list[str] = sorted(snap_keys - vault_keys)
    will_delete_all: list[str] = sorted(vault_keys - snap_keys)
    common = snap_keys & vault_keys

    will_overwrite_all: list[str] = []
    unchanged_count = 0
    for key in sorted(common):
        vault_sha = _file_sha256(vault_files[key])
        snap_sha = hashlib.sha256(snapshot_files[key]).hexdigest()
        if vault_sha != snap_sha:
            will_overwrite_all.append(key)
        else:
            unchanged_count += 1

    # Apply sample_limit
    total_changes = len(will_create_all) + len(will_delete_all) + len(will_overwrite_all)
    truncated = total_changes > sample_limit

    will_create = will_create_all[:sample_limit]
    will_overwrite = will_overwrite_all[:max(0, sample_limit - len(will_create))]
    will_delete = will_delete_all[:max(0, sample_limit - len(will_create) - len(will_overwrite))]

    # Extract metadata from snapshot name (format: YYYY-MM-DDTHH-MM-SS-<kind>)
    parts = snapshot_name.split("-")
    snap_timestamp = snapshot_name  # fallback
    snap_kind = "manual"
    if len(parts) >= 7:
        snap_kind = parts[-1] if parts[-1] in ("manual", "auto", "pre_restore") else "manual"

    return RestorePreview(
        snapshot_name=snapshot_name,
        snapshot_timestamp=snap_timestamp,
        snapshot_kind=snap_kind,
        snapshot_file_count=len(snap_keys),
        vault_file_count=len(vault_keys),

def remove_from_ignore(
    vault: Path,
    shas: list[str],
) -> tuple["LostSessionsIgnore", int]:
    ignore = load_ignore(vault)
    before = set(ignore.ignored)
    to_remove = set(shas)
    remaining = [s for s in ignore.ignored if s not in to_remove]
    removed = len(before) - len(remaining)
    ignore.ignored = remaining
    save_ignore(vault, ignore)
    return ignore, removed
```

3. Add `list_ignored_session_details`:

```python
def list_ignored_session_details(
    vault: Path,
    *,
    transcripts_root: Path | None = None,
) -> list[IgnoredSessionDetails]:
    ignore = load_ignore(vault)
    if not ignore.ignored:
        return []
    results: list[IgnoredSessionDetails] = []
    for sha in ignore.ignored:
        detail = IgnoredSessionDetails(sha=sha)
        # Attempt to find transcript for metadata
        if transcripts_root is not None:
            for p in transcripts_root.rglob(f"{sha}*.jsonl"):
                try:
                    detail.transcript_path = p.as_posix()
                    stat = p.stat()
                    detail.size_bytes = stat.st_size
                    import datetime as _dt
                    detail.mtime = _dt.datetime.fromtimestamp(
                        stat.st_mtime, tz=_dt.timezone.utc
                    ).isoformat()
                except OSError:
                    pass
                break
        results.append(detail)
    return results
```

Create `frontend/src/pages/IgnoredSessions.tsx`:

```typescript
import { useState } from "react";
import { useTranslation } from "react-i18next";
import { Checkbox } from "@/components/ui/checkbox";
import { Button } from "@/components/ui/button";
import { DaemonDownAlert } from "@/components/widgets/DaemonDownAlert";
import { ConfirmDialog } from "@/components/widgets/ConfirmDialog";
import { useIgnoredSessions, useUnIgnoreLostSessions } from "@/hooks/useIgnoredSessions";
import type { IgnoredSession } from "@/types/LostSession";

export default function IgnoredSessions() {
  const { t } = useTranslation();
  const { data, isLoading, isError, error } = useIgnoredSessions();
  const unIgnoreMut = useUnIgnoreLostSessions();
  const [selected, setSelected] = useState<Set<string>>(new Set());
  const [confirmOpen, setConfirmOpen] = useState(false);

  if (isError) return <DaemonDownAlert error={error} />;

  const sessions: IgnoredSession[] = data?.ignored ?? [];

  const toggleAll = () => {
    if (selected.size === sessions.length) {
      setSelected(new Set());
    } else {
      setSelected(new Set(sessions.map((s) => s.sha)));
    }
  };

  const toggle = (sha: string) => {
    const next = new Set(selected);
    if (next.has(sha)) next.delete(sha);
    else next.add(sha);
    setSelected(next);
  };

  const handleUnIgnore = () => {
    // Group by project
    const byProject: Record<string, string[]> = {};
    for (const s of sessions) {
      if (!selected.has(s.sha)) continue;
      const proj = s.project_name || "";
      byProject[proj] = [...(byProject[proj] ?? []), s.sha];
    }
    Promise.all(
      Object.entries(byProject).map(([project_name, shas]) =>
        unIgnoreMut.mutateAsync({ project_name, shas }),
      ),
    ).then(() => {
      setSelected(new Set());
      setConfirmOpen(false);
    });
  };

  return (
    <div className="p-6 space-y-4">
      <div className="flex items-center justify-between">
        <h1 className="text-xl font-semibold">{t("sections.ignored_sessions")}</h1>
        {selected.size > 0 && (
          <Button
            variant="outline"
            size="sm"
            onClick={() => setConfirmOpen(true)}
          >
            {t("ignored_sessions.unignore_selected", { count: selected.size })}
          </Button>
        )}
      </div>

      {isLoading && <p className="text-muted-foreground">{t("common.loading")}</p>}

      {!isLoading && sessions.length === 0 && (
        <p className="text-muted-foreground">{t("ignored_sessions.empty")}</p>
      )}

      {sessions.length > 0 && (
        <div className="space-y-2">
          <div className="flex items-center gap-2 text-sm text-muted-foreground">
            <Checkbox
              checked={selected.size === sessions.length && sessions.length > 0}
              onCheckedChange={toggleAll}
            />
            <span>{t("ignored_sessions.select_all")}</span>
          </div>
          {sessions.map((s) => (
            <div
              key={s.sha}
              className="flex items-center gap-3 p-3 rounded border bg-card"
            >
              <Checkbox
                checked={selected.has(s.sha)}
                onCheckedChange={() => toggle(s.sha)}
              />
              <div className="flex-1 min-w-0">
                <p className="text-sm font-mono truncate">{s.sha.slice(0, 12)}</p>
                {s.cwd && (
                  <p className="text-xs text-muted-foreground truncate">{s.cwd}</p>
                )}
                {s.project_name && (
                  <p className="text-xs text-muted-foreground">{s.project_name}</p>
                )}
              </div>
            </div>
          ))}
        </div>
      )}

      <ConfirmDialog
        open={confirmOpen}
        onOpenChange={setConfirmOpen}
        title={t("ignored_sessions.unignore_confirm_title")}
        description={t("ignored_sessions.unignore_confirm_body", {
          count: selected.size,
        })}
        onConfirm={handleUnIgnore}
      />
    </div>
  );
}
```
